Building a gaming-bot with a custom neural network: Part 1¶

Over the last couple of projects I realized that the coding best practices were not being applied in these notebooks. I focused on getting the job done without caring about readability/code maintenance. The way the scripts were designed were very heratic as well. In this notebook I'll focus on getting more "object-oriented" as it is conceptually easier for a someone to understand the code when it is uses classes and other common conventions. This notebook also tries to emulate a Java code structure by having clearly defined methods, objects and a main function at the end.

Now for the fun part,

I always wondered if I could build a bot for a game. The obvious choice for a mobile game would be to install Mobizen to mirror the screen on my PC and run a selenium script on my computer to automate the tasks. You'd be forgiven for thinking it was as simple as taking a screenshot of the situation, locate a couple of items and take a decision. It is not that simple.

I quickly hit a wall because most of the time selenium is not consistent at all. Selenium would sometimes not find the item on the screen mainly because it is too small or throw exceptions that need to be handled.

The solution to this problem came after taking Andrew Ng's free coursera class on machine learning. Given a dataset of labeled data I could simply train a neural net on a dataset and use the weights it found to classify new data.

A big thank you @John Wittenauer for the ML code and @ Andrew NG's explanations without whom this project would not have been possible.

The 3 datasets used in this notebook can be found on Github. The Raw pictures are part of our gaming bot, the ex2data1.txt and ex3data1.mat are both part of Andrew NG's ML exercises on coursera

def png_to_array(liste):
    df = pd.DataFrame()
    # Transforming .png images to array
    for counter, image in enumerate(liste):
        image_path = os.path.join(os.getcwd(),image) +'.png'
        image_instance = Image.open(image_path)
        image_array = np.asarray(image_instance) #np array of our image instance
        id_ = image[:image.find('-')] # unique id of our image
        pixel_dic = {}
        image_dic = {}
        for row, eachRow in enumerate(image_array):
            for pixel, eachPix in enumerate(eachRow):
                for pixel_color, color in enumerate(eachPix):
                    cpix = 'pixel' + str(row) + str(pixel) + str(pixel_color)
                    pixel_dic[cpix]=color # gradually append pixel information from a given image 
        image_dic[int(id_)] = pixel_dic # contains all pixel information of that image 
        ser = pd.DataFrame.from_dict(image_dic,orient='index') # transform dictionary into a series
        if counter == 0:
            print("initializing the dataframe")
            df = pd.DataFrame(columns=pixel_dic.keys()) #initialize the output file that we'll gradually concatenate to with the image data
            df = pd.concat([df, ser], axis=0, join_axes=[df.columns])
        else:
            df = pd.concat([df, ser], axis=0, join_axes=[df.columns])
            os.system('cls')
            print("Transforming .png images to array: ",counter+1,"/",len(liste))
    return df

def classify_rarity_image(image):
    final_list = []
    legend_count = 0
    rare_count = 0
    hero_count = 0
    iar = np.asarray(image[1])
    name = image[0]
    j =  name[name.find('-')+1:]
    rarity = j[:j.find('-')] # rarity of the rune
    for eachRow in iar:
        for eachPix in eachRow:
            if int(eachPix[1]) <= 120  & int(eachPix[1]) >= 80 & int(eachPix[2]) >= 170 & int(eachPix[2]) <= 200: ## get legend (orange)
                legend_count+=1
            if int(eachPix[1]) <= 120 & int(eachPix[2]) >= 170 & int(eachPix[2]) <= 200: ## get rare (blue)
                rare_count+=1
            if int(eachPix[1]) <= 100 & int(eachPix[2]) >= 100 & int(eachPix[0]) >= 100: ## get hero (purple)
                hero_count+=1
    final_list.append((image, rarity, legend_count, rare_count, hero_count))
    rune_rarity = max(final_list,key=lambda item:item[1])[1]
    # print("image name: ",rarity, "\nprediction: ", rune_rarity)
    return rarity == rune_rarity

def classify_rarity_images():
    counter = 0
    images = create_image_instance_list()
    for image in images:
        if classify_rarity_image(image) is True:
            counter+=1
    print(counter/len(images)*100,'% of images were correctly predicted')

class Model:
    def __init__(self, X, y, y_ravel, theta):
        self.X = X
        self.y = y

        self.theta = theta

    def log_reg(self):
        print_counter = 0
        def sigmoid(z):  
            return 1 / (1 + np.exp(-z))
        def cost(theta, X, y):      
            if print_counter == 0:
                print("Running the Logistic Regression on:\nShapes: X",X.shape,"y",y.shape,"theta",theta.shape)
                print_counter += 1
            theta = np.matrix(theta)
            X = np.matrix(X)
            y = np.matrix(y).astype(float)
            first = np.multiply(-y, np.log(sigmoid(X * theta.T)))
            second = np.multiply((1 - y), np.log(1 - sigmoid(X * theta.T)))
            return np.sum(first - second) / (len(X))
        def gradient(theta, X, y):  
            theta = np.matrix(theta)
            X = np.matrix(X)
            y = np.matrix(y).astype(float)
            parameters = int(theta.ravel().shape[1])
            grad = np.zeros(parameters)
            error = sigmoid(X * theta.T) - y
            for i in range(parameters):
                term = np.multiply(error, X[:,i])
                grad[i] = np.sum(term) / len(X)
                return grad   
        def predict(theta, X):  
            probability = sigmoid(X * theta.T)
            return [1 if x >= 0.5 else 0 for x in probability]
        optimized = opt.fmin_tnc(func=cost, x0=self.theta, fprime=gradient, args=(self.X, self.y)) 
        cost(optimized[0], self.X, self.y)
        theta_min = np.matrix(optimized[0])  
        predictions = predict(theta_min, self.X)  # list object
        correct = [1 if ((a == 1 and b == 1) or (a == 0 and b == 0)) else 0 for (a, b) in zip(predictions, self.y)]  # y is an np array with the target values
        print((sum(correct), len(correct)) )
        accuracy = (sum(map(int, correct)) / len(correct)) *100 
        print('log_reg accuracy = {0}%'.format(accuracy))
        return accuracy

def neural_net(X,y, hidden_size,num_labels,learning_rate):
    def sigmoid(z):  
        return 1 / (1 + np.exp(-z))
    def forward_propagate(X, theta1, theta2): 
        # print("forward_propagate: X",X.shape, "theta1",theta1.shape, "theta2",theta2.shape)
        m = X.shape[0]
        a1 = np.insert(X, 0, values=np.ones(m), axis=1)
        z2 = a1 * theta1.T
        a2 = np.insert(sigmoid(z2), 0, values=np.ones(m), axis=1)
        z3 = a2 * theta2.T
        h = sigmoid(z3)
        return a1, z2, a2, z3, h
    # unused function for now
    def cost(params, input_size, hidden_size, num_labels, X, y, learning_rate):  
        m = X.shape[0]
        X = np.matrix(X)
        y = np.matrix(y)
        # reshape the parameter array intQo parameter matrices for each layer
        theta1 = np.matrix(np.reshape(params[:hidden_size * (input_size + 1)], (hidden_size, (input_size + 1))))
        theta2 = np.matrix(np.reshape(params[hidden_size * (input_size + 1):], (num_labels, (hidden_size + 1))))
        # run the feed-forward pass
        a1, z2, a2, z3, h = forward_propagate(X, theta1, theta2)
        # compute the cost
        J = 0
        for i in range(m):
            first_term = np.multiply(-y[i,:], np.log(h[i,:]))
            second_term = np.multiply((1 - y[i,:]), np.log(1 - h[i,:]))
            J += np.sum(first_term - second_term)
        J = J / m
        return J
    def sigmoid_gradient(z):  
        return np.multiply(sigmoid(z), (1 - sigmoid(z)))
    def backprop(params, input_size, hidden_size, num_labels, X, y, learning_rate):  
        ##### this section is identical to the cost function logic we already saw #####
        m = X.shape[0]
        X = np.matrix(X)
        y = np.matrix(y)
        # reshape the parameter array into parameter matrices for each layer
        theta1 = np.matrix(np.reshape(params[:hidden_size * (input_size + 1)], (hidden_size, (input_size + 1))))
        theta2 = np.matrix(np.reshape(params[hidden_size * (input_size + 1):], (num_labels, (hidden_size + 1))))
        # run the feed-forward pass
        a1, z2, a2, z3, h = forward_propagate(X, theta1, theta2)
        # initializations
        J = 0
        delta1 = np.zeros(theta1.shape)  # (25, 401)
        delta2 = np.zeros(theta2.shape)  # (10, 26)
        # compute the cost
        for i in range(m):
            first_term = np.multiply(-y[i,:], np.log(h[i,:]))
            second_term = np.multiply((1 - y[i,:]), np.log(1 - h[i,:]))
            J += np.sum(first_term - second_term)
        J = J / m
        # add the cost regularization term
        J += (float(learning_rate) / (2 * m)) * (np.sum(np.power(theta1[:,1:], 2)) + np.sum(np.power(theta2[:,1:], 2)))
        ##### end of cost function logic, below is the new part #####
        # perform backpropagation
        for t in range(m):
            a1t = a1[t,:]  # (1, 401)
            z2t = z2[t,:]  # (1, 25)
            a2t = a2[t,:]  # (1, 26)
            ht = h[t,:]  # (1, 10)
            yt = y[t,:]  # (1, 10)
            d3t = ht - yt  # (1, 10)
            z2t = np.insert(z2t, 0, values=np.ones(1))  # (1, 26)
            d2t = np.multiply((theta2.T * d3t.T).T, sigmoid_gradient(z2t))  # (1, 26)
            delta1 = delta1 + (d2t[:,1:]).T * a1t
            delta2 = delta2 + d3t.T * a2t
        delta1 = delta1 / m
        delta2 = delta2 / m
        # add the gradient regularization term
        delta1[:,1:] = delta1[:,1:] + (theta1[:,1:] * learning_rate) / m
        delta2[:,1:] = delta2[:,1:] + (theta2[:,1:] * learning_rate) / m
        # unravel the gradient matrices into a single array
        grad = np.concatenate((np.ravel(delta1), np.ravel(delta2)))
        return J, grad
    encoder = OneHotEncoder(sparse=False)  
    y_onehot = encoder.fit_transform(y)  
    # print('y_onehot.shape',y_onehot.shape)
    input_size = X.shape[1]
    # randomly initialize a parameter array of the size of the full network's parameters
    params = (np.random.random(size=hidden_size * (input_size + 1) + num_labels * (hidden_size + 1)) - 0.5) * 0.25
    X = np.matrix(X)  
    y = np.matrix(y)
    # unravel the parameter array into parameter matrices for each layer
    theta1 = np.matrix(np.reshape(params[:hidden_size * (input_size + 1)], (hidden_size, (input_size + 1))))  
    theta2 = np.matrix(np.reshape(params[hidden_size * (input_size + 1):], (num_labels, (hidden_size + 1))))
    #X.shape, theta1.shape, theta2.shape # 103, 40264     25x401    10x26 vs 5000,400
    print("Initializing NN...")
    a1, z2, a2, z3, h = forward_propagate(X, theta1, theta2)  
    # print('a1.',a1,'\nz2.', z2,'\na2.', a2, '\nz3.', z3, '\nh.', h)
    print('a1.shape',a1.shape,'\nz2.shape', z2.shape,'\na2.shape', a2.shape, '\nz3.shape', z3.shape, '\nh.shape', h.shape)
    print("Backpropagation...")
    J, grad = backprop(params, input_size, hidden_size, num_labels, X, y_onehot, learning_rate)  
    print('J (Cost):',J, 'grad.shape:', grad.shape)
    # minimize the objective function
    fmin = minimize(fun=backprop, x0=params, args=(input_size, hidden_size, num_labels, X, y_onehot, learning_rate),  
                    method='TNC', jac=True, options={'maxiter': 250})
    print('fmin',fmin)  
    X = np.matrix(X)  
    theta1 = np.matrix(np.reshape(fmin.x[:hidden_size * (input_size + 1)], (hidden_size, (input_size + 1))))  
    theta2 = np.matrix(np.reshape(fmin.x[hidden_size * (input_size + 1):], (num_labels, (hidden_size + 1))))
    print("theta1",theta1, "theta2", theta2)
    print("theta1.shape",theta1.shape, "theta2.shape", theta2.shape)
    print("Feed-forward with the optimized thetas...")
    a1, z2, a2, z3, h = forward_propagate(X, theta1, theta2)  
    # y_pred = np.array(np.argmax(h, axis=1) + 1)  
    y_pred = np.array(np.argmax(h, axis=1)) 
    print('y_pred',y_pred) 
    correct = [1 if a == b else 0 for (a, b) in zip(y_pred, y.astype(int))]  
    accuracy = (sum(map(int, correct)) / float(len(correct)))  
    print('accuracy = {0}%'.format(accuracy * 100))
    return accuracy, correct, y_pred, y, X, a1, z2, a2, z3, h, theta1, theta2, J, grad

def forward_propagate(X, theta1, theta2): 
    # print("forward_propagate: X",X.shape, "theta1",theta1.shape, "theta2",theta2.shape)
    m = X.shape[0]
    a1 = np.insert(X, 0, values=np.ones(m), axis=1)
    z2 = a1 * theta1.T
    a2 = np.insert(sigmoid(z2), 0, values=np.ones(m), axis=1)
    z3 = a2 * theta2.T
    h = sigmoid(z3)
    return a1, z2, a2, z3, h

def sigmoid_gradient(z):  
    return np.multiply(sigmoid(z), (1 - sigmoid(z)))

def sigmoid(z):  
    return 1 / (1 + np.exp(-z))
